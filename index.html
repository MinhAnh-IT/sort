Code: Quick Sort (Python)
def quick_sort(arr, lo, hi): if lo < hi: p = partition(arr, lo, hi) quick_sort(arr, lo, p - 1) quick_sort(arr, p + 1, hi)   def partition(arr, lo, hi): pivot = arr[hi] # chọn phần tử cuối i = lo - 1 # ranh giới vùng ≤ pivot for j in range(lo, hi): if arr[j] <= pivot: i += 1 arr[i], arr[j] = arr[j], arr[i] arr[i + 1], arr[hi] = arr[hi], arr[i + 1] return i + 1
